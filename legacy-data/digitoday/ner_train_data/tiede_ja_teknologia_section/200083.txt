<DATE>

<INGRESS>

,

<BODY>
<PARAGRAPH>

.

.

.

<PARAGRAPH>
IBM:n

,

"
taltan
"
,

,

.

<PARAGRAPH>
Tekniikalle

,

.

Keksintöä

.

<PARAGRAPH>

,

.

usein
