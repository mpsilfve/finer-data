<INGRESS>

,

Ison-Britannian

.

.

<BODY>
<PARAGRAPH>
Ison-Britannian

,

.

Hallituksen

(

)

eli

.

<PARAGRAPH>
Haktivismi
on

,

Anonymous-liiga
on
usein

.

<PARAGRAPH>
Britannian

GCHQ:n

iskien

.

PowerPoint-esitys
vuodelta
2012

,

osan

.

"

on

"

ja

.

Erikseen

,

,

,

.

<PARAGRAPH>

,

"

"

.

Tietoturvayhtiö

,

.

<PARAGRAPH>

,

.

<PARAGRAPH>

.
