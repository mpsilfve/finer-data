.

.

<PARAGRAPH>
–

,

,

.

<PARAGRAPH>

.

<PARAGRAPH>
Pörssissä

,

.

Lopputuloksena

99

TRS
:
n

.

<PARAGRAPH>
Yksi
TRS
:
n

,

.

Yhdessä

73
prosenttia
TRS
:
n
osakkeista
.

<PARAGRAPH>
Dotcomin
mukaan

,

.

<PARAGRAPH>
Kaupan

:

,

.

<PARAGRAPH>
Suunnitelman

.

TRS
:
n

.

<PARAGRAPH>

,

.

.

<PARAGRAPH>

.

.
