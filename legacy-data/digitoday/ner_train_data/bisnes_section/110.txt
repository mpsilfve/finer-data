.

.

<PARAGRAPH>

,

.

,

.

<PARAGRAPH>

.

,

.

,

.

Keskustelukulttuuri

.

<PARAGRAPH>
Julkinen

.

Pelialan

,

oli

.

Sitä

,

.

<PARAGRAPH>
Nokia-vertaus
on

,

.

Yhtiöillä

.

<PARAGRAPH>
Molemmista

,

.

<PARAGRAPH>

.

<PARAGRAPH>
Kun

,

–

.

<PARAGRAPH>
Asetelmasta

,

,

.

.

,

.

Vaikka

,

.

<PARAGRAPH>
Äkäinen

USA:ssa

.

.

,

.

<PARAGRAPH>

,

.

.

.
